			x' = \pm \sqrt{1-x^2}
		\end{equation*}

		\begin{equation*}
		x'' = \frac{|x|}{\sqrt{1-x^2}}.
		\end{equation*}
		Taking the limit as $x$ approaches $-1$ yields:
		\begin{equation*}
		\lim_{x \to -1} |x''| = \frac{1}{\sqrt{1 - 1}} = \frac{1}{0} = \infty.
		\end{equation*}
		Since $|x'|$ is continuous on $(-1,1)$ and is not lipschitz then $(1)$ has infinite solutions.
